// ─── Task 完成回调（核心）───

export async function onTaskCompleted(taskId: string): Promise<void> {
  const task = taskStore.getTaskById(taskId);
  if (!task?.autodataMeta) return;

  const { role, iterationGroupId } = task.autodataMeta;
  const group = activeGroups.get(iterationGroupId);
  if (!group) return;

  switch (role) {
    case "challenger":
      await handleChallengerDone(task, group);
      break;
    case "weak_solver":
      group.weakDone = true;
      await checkBothSolversDone(task, group);
      break;
    case "strong_solver":
      group.strongDone = true;
      await checkBothSolversDone(task, group);
      break;
    case "judge":
      handleJudgeDone(task, group);
      break;
  }
}

## Phase 2：迭代精炼闭环

### 目标

验证不通过时自动重新生成，直到通过或达到最大轮次。

### 新增/修改

- `server/services/autodataService.ts` — 扩展迭代循环逻辑
- `server/store/autodataStore.ts` — 新增迭代组持久化存储
- `server/routes/autodata.ts` — 新增手动重试端点

### 核心逻辑

```
loop (maxIterations = 5):
  1. Challenger 生成 QA + Rubric
  2. Weak Solver 回答
  3. Strong Solver 回答
  4. Judge 按 Rubric 评分
  5. 计算 gap = strongAvg - weakAvg
  6. If gap ≥ 20% && weak ≤ 65% && strong ≥ 60%:
       → ACCEPTED
     Else:
       → 分析失败原因 → 注入反馈到下一轮 Challenger prompt
```

### 元优化流程

```
1. 收集 Autodata 任务完整轨迹
2. 定期运行"分析 Agent"读取失败轨迹
3. 识别系统性模式（如"上下文泄露"、"问题太泛"）
4. 基于分析结果生成 prompt 改进建议
5. 在验证集上 A/B 测试新旧 prompt
6. 只保留提升通过率的改进
```

## 风险清单

| 风险 | 等级 | 降级方案 |
|------|------|---------|
| Judge 输出格式不稳定 | 中 | structured output 约束 + fallback 正则解析 |
| 迭代循环中断 | 中 | 超时检测 + 手动重试端点 |
| 编排 Agent 上下文溢出 | 中 | 限制 maxIterations=3；每轮只保留摘要 |
| Token 消耗过高 | 低 | Phase 1 先做单轮验证；降低 solver 调用次数 |
| JSON 存储不适合大量轨迹 | 低 | Phase 4 使用 JSONL 追加写入 |
