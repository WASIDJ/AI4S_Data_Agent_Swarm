# Agent Swarm 安全认证方案

> 版本：v1.2  
> 日期：2026-05-13  
> 状态：已确认

## 1. 概述

### 1.1 设计目标

| 阶段 | 场景 | 用户模型 | 认证方式 |
|------|------|---------|---------|
| 当前 | 本地单用户 | 默认管理员账户 | JWT Token |
| 未来 | 服务端部署 | 多用户 + 双角色 | JWT Token |

### 1.2 核心原则

- **所有后端接口强制认证**：除登录/健康检查外，全部需要 Token
- **数据隔离**：用户只能访问自己的数据（通过 userId 过滤）
- **角色扩展**：预留 admin/user 角色，支持未来权限控制

## 3. 认证机制

### 3.1 登录流程

```
POST /api/auth/login
请求: { account: "admin", password: "admin123" }
响应: { token: "eyJhbGciOiJ...", user: { id, name, email, role } }
```

### 3.2 Token 规范

| 配置项 | 值 |
|--------|-----|
| 签名算法 | HS256 |
| Secret | `JWT_SECRET`（环境变量，必填） |
| 过期时间 | 7 天 |
| Payload | `{ userId, email, role }` |

### 3.3 Token 验证

每个请求必须携带：
```
Authorization: Bearer <token>
```

### 4.2 认证中间件

```typescript
// middleware/auth.ts

import { Request, Response, NextFunction } from "express";
import jwt from "jsonwebtoken";

const JWT_SECRET = process.env.JWT_SECRET || "ai4s-swarm-local-dev-secret-2026";

export interface JwtPayload {
  userId: string;
  email: string;
  role: string;
}

declare global {
  namespace Express {
    interface Request {
      user?: JwtPayload;
    }
  }
}

// 强制认证中间件
export function requireAuth(req: Request, res: Response, next: NextFunction): void {
  const authHeader = req.headers.authorization;
  const token = authHeader?.replace("Bearer ", "");
  
  if (!token) {
    res.status(401).json({ 
      error: { code: "UNAUTHORIZED", message: "请先登录" } 
    });
    return;
  }
  
  try {
    req.user = jwt.verify(token, JWT_SECRET) as JwtPayload;
    next();
  } catch {
    res.status(401).json({ 
      error: { code: "UNAUTHORIZED", message: "Token 无效或已过期" } 
    });
    return;
  }
}

// 管理员权限（未来扩展）
export function requireAdmin(req: Request, res: Response, next: NextFunction): void {
  if (!req.user) {
    res.status(401).json({ error: { code: "UNAUTHORIZED", message: "请先登录" } });
    return;
  }
  if (req.user.role !== "admin") {
    res.status(403).json({ error: { code: "FORBIDDEN", message: "需要管理员权限" } });
    return;
  }
  next();
}

// Token 签发
export function signToken(payload: JwtPayload): string {
  return jwt.sign(payload, JWT_SECRET, { expiresIn: "7d" });
}
```

## 5. 数据隔离机制

### 5.1 设计思路

**方案对比**：

| 方案 | 优点 | 缺点 |
|------|------|------|
| 修改现有模型 | 查询简单直接 | 改动大、需迁移数据、侵入性强 |
| 映射表（✅ 采用） | 不改现有模型、改动最小 | 查询需 join |

**采用映射表方案**：新增独立的资源归属映射表，不修改现有数据模型。

### 5.2 映射表设计

```typescript
// 新增：资源归属映射表
interface ResourceOwnership {
  id: string;              // 映射记录 ID
  userId: string;          // 用户 ID
  resourceType: string;    // 资源类型："project" | "agent" | "task"
  resourceId: string;      // 资源 ID
  createdAt: number;       // 创建时间
}

// 存储文件：data/ownerships.json
interface OwnershipsEnvelope {
  _schema_version: number;
  ownerships: ResourceOwnership[];
}
```

### 5.3 映射关系

```
User (id)
  ↓ N:M (通过 ResourceOwnership)
Project (id)  ← resourceType: "project"
Agent (id)    ← resourceType: "agent"
Task (id)     ← resourceType: "task"
```

### 5.4 Store 实现

```typescript
// store/ownershipStore.ts

import { ownershipsStore } from "./index.js";

const ownerships = new Map<string, ResourceOwnership>();

// 创建归属关系
export function grantOwnership(
  userId: string,
  resourceType: "project" | "agent" | "task",
  resourceId: string,
): ResourceOwnership {
  const ownership: ResourceOwnership = {
    id: crypto.randomUUID(),
    userId,
    resourceType,
    resourceId,
    createdAt: Date.now(),
  };
  ownerships.set(ownership.id, ownership);
  persist();
  return ownership;
}

// 查询用户的资源 ID 列表
export function getUserResourceIds(
  userId: string,
  resourceType: "project" | "agent" | "task",
): string[] {
  return Array.from(ownerships.values())
    .filter(o => o.userId === userId && o.resourceType === resourceType)
    .map(o => o.resourceId);
}

// 查询资源的归属用户
export function getResourceOwner(
  resourceType: "project" | "agent" | "task",
  resourceId: string,
): string | undefined {
  const ownership = Array.from(ownerships.values())
    .find(o => o.resourceType === resourceType && o.resourceId === resourceId);
  return ownership?.userId;
}

// 检查用户是否有权访问资源
export function hasAccess(
  userId: string,
  resourceType: "project" | "agent" | "task",
  resourceId: string,
): boolean {
  return Array.from(ownerships.values())
    .some(o => 
      o.userId === userId && 
      o.resourceType === resourceType && 
      o.resourceId === resourceId
    );
}

// 删除资源时清理映射
export function revokeOwnership(
  resourceType: "project" | "agent" | "task",
  resourceId: string,
): void {
  for (const [id, o] of ownerships) {
    if (o.resourceType === resourceType && o.resourceId === resourceId) {
      ownerships.delete(id);
    }
  }
  persist();
}

function persist(): void {
  ownershipsStore.save({
    _schema_version: 1,
    ownerships: Array.from(ownerships.values()),
  });
}

export function loadOwnerships(): void {
  const data = ownershipsStore.getData();
  const list = (data.ownerships as ResourceOwnership[]) ?? [];
  ownerships.clear();
  for (const o of list) {
    ownerships.set(o.id, o);
  }
}
```

### 5.7 访问控制

```typescript
// 通用权限检查中间件
export function requireResourceAccess(
  resourceType: "project" | "agent" | "task",
  idParam: string = "id",
) {
  return (req: Request, res: Response, next: NextFunction): void => {
    const resourceId = req.params[idParam];
    
    // 管理员可访问所有资源
    if (req.user?.role === "admin") {
      return next();
    }
    
    // 检查归属
    const hasAccess = ownershipStore.hasAccess(
      req.user!.userId,
      resourceType,
      resourceId,
    );
    
    if (!hasAccess) {
      return res.status(403).json({ 
        error: { code: "FORBIDDEN", message: "无权访问此资源" } 
      });
    }
    
    next();
  };
}

// 使用示例
projectsRouter.get("/:id", requireAuth, requireResourceAccess("project"), (req, res) => {
  // 已通过权限检查
  const project = getProjectById(req.params.id);
  res.json({ code: 0, data: project });
});

agentsRouter.delete("/:id", requireAuth, requireResourceAccess("agent"), (req, res) => {
  const agentId = req.params.id;
  deleteAgent(agentId);
  ownershipStore.revokeOwnership("agent", agentId);  // 清理映射
  res.json({ code: 0, message: "删除成功" });
});
```

## 7. 前端适配

### 7.1 登录流程

```typescript
// 登录页面
async function login(account: string, password: string) {
  const res = await api.post("/api/auth/login", { account, password });
  if (res.data.code === 0) {
    localStorage.setItem("token", res.data.data.token);
    localStorage.setItem("user", JSON.stringify(res.data.data.user));
    router.push("/");
  }
}
```

### 7.2 请求拦截

```typescript
// api/client.ts
import axios from "axios";

const api = axios.create({ baseURL: "/api" });

// 自动携带 Token
api.interceptors.request.use(config => {
  const token = localStorage.getItem("token");
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// 401 自动跳转登录
api.interceptors.response.use(
  res => res,
  err => {
    if (err.response?.status === 401) {
      localStorage.removeItem("token");
      localStorage.removeItem("user");
      window.location.href = "/login";
    }
    return Promise.reject(err);
  }
);
```

### 7.3 启动检查

```typescript
// main.ts 或 App.tsx

if (!token && !isPublicPage()) {
  window.location.href = "/login";
}
```
