    START(["🚀 用户创建 Pipeline"]) --> C["Challenger

    C -- "写入 round_N_challenger.json" --> WAIT["⏳ 等待完成"]
    WAIT --> WS["Weak Solver
回答问题"]
    WAIT --> SS["Strong Solver
高质量回答"]
    WS -- "写入 round_N_weak_solver.json" --> JOIN["⏳ 两个 Solver 都完成"]
    SS -- "写入 round_N_strong_solver.json" --> JOIN
    JOIN --> J["Judge

    J -- "写入 round_N_judge.json" --> EVAL{"📊 计算评分
weakScore ≤ 65%
strongScore ≥ 60%
gap ≥ 20%"}

    EVAL -- "✅ 通过" --> ACCEPT(["🎉 ACCEPTED"])
    EVAL -- "❌ 未通过" --> CHECK{"达到 maxRounds?"}
    CHECK -- "是" --> REJECT(["⛔ REJECTED"])
    CHECK -- "否" --> FEEDBACK["📝 分析失败原因
注入反馈"]
    FEEDBACK --> C

sequenceDiagram
    participant U as 用户
    participant API as /api/autodata
    participant SVC as autodataService
    participant TM as taskManager
    participant SM as sdkSessionManager
    participant Agent as Agent (各角色)

    U->>API: POST /create
    API->>SVC: createAutodataPipeline()
    SVC->>TM: startTask(Challenger)
    TM->>SM: startTask()
    SM->>Agent: Challenger 生成 QA+Rubric
    Agent-->>SM: stream 完成
    SM->>SVC: onTaskCompleted(challenger)
    Note over SVC: handleChallengerDone
    SVC->>TM: startTask(Weak Solver)
    SVC->>TM: startTask(Strong Solver)
    Note over Agent: 两个 Solver 并行执行
    SM->>Agent: Weak Solver 回答
    SM->>Agent: Strong Solver 回答
    SM->>SVC: onTaskCompleted(solver) ×2
    Note over SVC: handleSolverDone (两次)
    SVC->>TM: startTask(Judge)
    SM->>Agent: Judge 评分
    SM->>SVC: onTaskCompleted(judge)
    Note over SVC: handleJudgeDone
    SVC->>SVC: 计算 scores, 判断通过?
    alt 通过
        SVC-->>U: WebSocket: autodata_accepted
    else 未通过且有剩余轮次
        SVC->>TM: startTask(下一轮 Challenger)
        Note over SVC: 注入反馈到 prompt
    else 达到 maxRounds
        SVC-->>U: WebSocket: autodata_rejected
    end
